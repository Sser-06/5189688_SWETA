#include<iostream>
#include<vector>
#include<cmath>
using namespace std;

int diagonalDifference(const vector<vector<int>>& matrix){
    int n = matrix.size();
    int primary = 0, secondary =0;
    
    for(int i=0; i<n; ++i){
        primary= primary +matrix[i][i];
        secondary = secondary + matrix[i][n-i-1];
    }
    
    return abs(primary - secondary);
}

int main(){
    int n;
    cin>>n;
    
    vector<vector<int>>arr(n,vector<int>(n));
    
    for(int i=0; i<n; ++i)
      for(int j=0;j<n;++j)
      cin>>arr[i][j];
    cout<<diagonalDifference(arr)<<endl;
      return 0;
}
