#include <iostream>
#include <vector>
#include <unordered_map>

using namespace std;
vector<int> matchStrings(vector<string>strings, vector<string> queries){
    unordered_map<string,int>freq;
    
    for(const auto& s: strings){
        freq[s]++;
    }
    
    vector<int> results;
    for(const auto& q: queries){
        if(freq.find(q)!= freq.end()){
            results.push_back(freq[q]);
        } else{
            results.push_back(0);
        }
    }
    return results;
}
int main(){
    int n;
    cin>>n;
    vector<string>strings(n);
    for(int i=0;i<n;++i){
        cin>>strings[i];
    }
    int q;
    cin>>q;
    vector<string>queries(q);
    for(int i=0;i<q;++i){
        cin>>queries[i];
    }
    
    vector<int>result = matchStrings(strings,queries);
   for(auto count : result){
    cout<<count<<"\n";
   }
   return 0;
}
