#include<iostream>
#include<string>
using namespace std;

string pangrams(string s){
    int alphabet[26] = {0};
    int count =0;
    for(int i=0;i<s.length();i++){
        char ch = s[i];
        
        if(ch>= 'A' && ch<='Z'){
            ch = ch + 32;
        }
        
        if(ch>='a' && ch<='z'){
            int index = ch -'a';
            if(alphabet[index]==0){
                alphabet[index]=1;
                count++;
            }
        }
    }
    
    if(count==26){
        return "pangram";
        
    } else{
        return "not pangram";
    }
}

int main(){
    string sentence;
    getline(cin,sentence);
    string result = pangrams(sentence);
    cout<<result<<endl;
    return 0;
}
