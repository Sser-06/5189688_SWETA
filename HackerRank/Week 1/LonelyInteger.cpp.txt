#include<iostream>
#include<vector>
using namespace std;

    int LonelyInteger(vector<int>a){
        int res =0;
        
        for(int num:a){
            res = res^num;
        }
        return res;
    }
    
    int main(){
        int n;
        cin>>n;
        vector<int> a(n);
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        int unique = LonelyInteger(a);
        cout<< unique<<"\n";
        return 0;
    }
